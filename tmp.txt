servers="$(grep SERVER: data/servers.txt | sed 's/[0-9] SERVER://')"
servers="$servers new"

PS3="Select Host: "
select server in $servers
do
  if [ $server="new" ]
  then
    add-server
  else
    break
  fi
  servers="$(grep SERVER: data/servers.txt | sed 's/[0-9] SERVER://')"
  servers="$servers new"
done
